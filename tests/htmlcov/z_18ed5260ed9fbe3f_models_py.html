170 def validate_element(self, new_date, date_to_validate=date.today()): 

171 """ 

172 Validates and updates a Todo element's planned_date if the new date is in the future. 

173 

174 This method checks if the provided date is later than the current date and 

175 updates the element's date accordingly. The changes are automatically saved 

176 to the database if the update is successful. 

177 

178 Args: 

179 new_date (date): The new date to validate and potentially set 

180 

181 Returns: 

182 bool: True if the date was updated, False if the new date is not later than current date 

183 """ 

184 if self.state in ("done", "cancel"): 

185 return False, _("Cette tâche est déjà terminée ou annulée.") 

186 

187 if new_date <= self.planned_date: 

188 return False, _("La date doit être postérieure à la date planifiée actuelle.") 

189 

190 self.planned_date = new_date 

191 self.report_date = None 

192 self.done_date = date_to_validate 

193 self.state = "todo" 

194 self.save() 

195 return True, ""

225 def report_element(self, date_of_report=date.today()): 

226 """ 

227 Reports the element to the user. 

228 

229 This method sets the element's state to "report" and updates the date to the next date. 

230 The changes are automatically saved to the database if the update is successful. 

231 """ 

232 if self.state != "done": 

233 self.planned_date = date_of_report + timedelta(days=1) 

234 self.state = "report" 

235 if self.report_date is None: 

236 self.report_date = date_of_report 

237 self.save()

239 def new_day(self, new_planned_date=date.today()): 

240 """ 

241 Updates the element's current date to now. 

242 Updates all planned dates to now. 

243 set the state to "report" if the element is not done. 

244 

245 This method updates the element's current date to the next day and saves the changes. 

246 

247 """ 

248 if self.state != "done" and self.planned_date < new_planned_date: 

249 self.planned_date = new_planned_date 

250 self.state = "report" 

251 if self.report_date is None: 

252 self.report_date = new_planned_date 

253 self.save()
